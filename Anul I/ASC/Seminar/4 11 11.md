```asm

	s db 'ana are un capac'
	l equ $ - s

	cuvant times l db 0
	invers times l db 0

segment code...
	start:

	mov ecx, l
	jecxz final
	cld
	mov esi, s
	mov edi, cuvant
	mov ebx, 0

	bucla:
		lodsb
		cmp al, ' '
		je urmatorul
		stosb
		inc ebx
		jmp next
		urmatorul:
			push ecx
			push esi
			push edi
			mov ecx, ebx
			mov esi, cuvant
			mov edi, invers
			add edi, ebx
			sub edi, 1
			inverseaza:
				cld
				lodsb
				std
				stosb
			loop inverseaza
			mov ecx, ebx
			mov esi, cuvant
			mov edi, invers
			repe cmpsb
			cmp ecx, 0
			je palindrom
			jne restaurare
		palindrom:
			inc edx
		restaurare:
			pop edi
			pop esi
			pop ecx
		next
		loop bucla
	final:
	...
```
